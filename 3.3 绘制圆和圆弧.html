var mycanvas=document.getElementById("canvas1");

var cntx=mycanvas.getContext('2d');

cntx.beginPath();

cntx.arc(200,200,100,0,Math.PI*2,true);

cntx.closePath();

cntx.lineWidth="4";//绘制轮廓时轮廓线宽度

cntx.strokeStyle="black";//设置绘图时的使用颜色

cntx.stroke();//cntx已经由路径画了圆的效果，这里直接调用stroke（）绘制
