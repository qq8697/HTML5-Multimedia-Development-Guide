var mygradient=cntx.createLinearGradient(0,100,400,300);

//添加颜色点

mygradient.addColorStop("0", "#ff0000");

mygradient.addColorStop("1.0", "#0000ff");

//填充图形

cntx.fillStyle=mygradient;

cntx.fillRect(0,0,400,400);
